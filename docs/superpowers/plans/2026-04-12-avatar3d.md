```python
import bpy, os

# ── Limpiar escena ────────────────────────────────────────────────────────────
bpy.ops.object.select_all(action='SELECT')
bpy.ops.object.delete(use_global=False)
for col in [bpy.data.meshes, bpy.data.materials, bpy.data.objects]:
    for item in list(col):
        col.remove(item, do_unlink=True)

# ── Factory de materiales (Principled BSDF, colores planos) ──────────────────
def mk_mat(name, hex_color):
    r = int(hex_color[1:3], 16) / 255
    g = int(hex_color[3:5], 16) / 255
    b = int(hex_color[5:7], 16) / 255
    m = bpy.data.materials.new(name)
    m.use_nodes = True
    bsdf = m.node_tree.nodes['Principled BSDF']
    bsdf.inputs['Base Color'].default_value = (r, g, b, 1.0)
    bsdf.inputs['Roughness'].default_value = 1.0
    bsdf.inputs['Metallic'].default_value = 0.0
    spec_key = 'Specular IOR Level' if 'Specular IOR Level' in bsdf.inputs else 'Specular'
    bsdf.inputs[spec_key].default_value = 0.0
    return m

M = {
    'skin':   mk_mat('skin',   '#B97A56'),
    'hair':   mk_mat('hair',   '#1A1008'),
    'hoodie': mk_mat('hoodie', '#1C1C26'),
    'pants':  mk_mat('pants',  '#2A3A5C'),
    'shoes':  mk_mat('shoes',  '#EEEEEE'),
    'eyes':   mk_mat('eyes',   '#111111'),
    'eyehl':  mk_mat('eyehl',  '#FFFFFF'),
}

# ── Helpers de geometría ──────────────────────────────────────────────────────
def sph(name, mat_key, loc, sx=1, sy=1, sz=1, segs=20, rings=14):
    bpy.ops.mesh.primitive_uv_sphere_add(
        radius=1, segments=segs, ring_count=rings, location=loc)
    o = bpy.context.active_object
    o.name = name
    o.scale = (sx, sy, sz)
    o.data.materials.clear()
    o.data.materials.append(M[mat_key])
    bpy.ops.object.shade_smooth()
    return o

def cyl(name, mat_key, loc, r=0.5, h=1.0, segs=14):
    bpy.ops.mesh.primitive_cylinder_add(
        radius=r, depth=h, vertices=segs, location=loc)
    o = bpy.context.active_object
    o.name = name
    o.data.materials.clear()
    o.data.materials.append(M[mat_key])
    bpy.ops.object.shade_smooth()
    return o

def cube(name, mat_key, loc, sx=1, sy=1, sz=1):
    bpy.ops.mesh.primitive_cube_add(location=loc)
    o = bpy.context.active_object
    o.name = name
    o.scale = (sx, sy, sz)
    o.data.materials.clear()
    o.data.materials.append(M[mat_key])
    return o

# ── HEAD GROUP (empty — para head tracking en Three.js) ──────────────────────
# Blender coordenadas: Z hacia arriba. Head center en Z=3.4.
bpy.ops.object.empty_add(type='PLAIN_AXES', location=(0, 0, 3.4))
head_grp = bpy.context.active_object
head_grp.name = 'head_group'

def parent(obj):
    bpy.context.view_layer.update()
    obj.parent = head_grp
    obj.matrix_parent_inverse = head_grp.matrix_world.inverted()

# ── CABEZA ────────────────────────────────────────────────────────────────────
head = sph('head', 'skin', (0, 0, 3.4), sx=0.90, sy=0.82, sz=1.0)
parent(head)

# ── CABELLO (cap) ─────────────────────────────────────────────────────────────
hair_top = sph('hair', 'hair', (0, 0, 3.72), sx=0.92, sy=0.84, sz=0.62)
parent(hair_top)
hairL = sph('hairL', 'hair', (-0.80, 0, 3.50), sx=0.36, sy=0.44, sz=0.54)
hairR = sph('hairR', 'hair', ( 0.80, 0, 3.50), sx=0.36, sy=0.44, sz=0.54)
parent(hairL); parent(hairR)

# ── OJOS (óvalos planos — esfera fuertemente comprimida en Y/profundidad) ─────
# En Blender Y es profundidad (hacia la cámara en viewport frontal)
eyeL = sph('eyeL', 'eyes', (-0.30, 0.76, 3.36), sx=0.24, sy=0.06, sz=0.18)
eyeR = sph('eyeR', 'eyes', ( 0.30, 0.76, 3.36), sx=0.24, sy=0.06, sz=0.18)
hlL  = sph('eyeHL_L', 'eyehl', (-0.20, 0.81, 3.42), sx=0.07, sy=0.04, sz=0.06)
hlR  = sph('eyeHL_R', 'eyehl', ( 0.38, 0.81, 3.42), sx=0.07, sy=0.04, sz=0.06)
parent(eyeL); parent(eyeR); parent(hlL); parent(hlR)

# ── CUELLO ────────────────────────────────────────────────────────────────────
cyl('neck', 'skin', (0, 0, 2.80), r=0.22, h=0.44, segs=12)

# ── CUERPO / HOODIE ───────────────────────────────────────────────────────────
cyl('body', 'hoodie', (0, 0, 1.95), r=0.62, h=1.50, segs=16)
# Hombros — silueta más ancha que el torso
sph('shoulderL', 'hoodie', (-0.76, 0, 2.50), sx=0.44, sy=0.38, sz=0.38)
sph('shoulderR', 'hoodie', ( 0.76, 0, 2.50), sx=0.44, sy=0.38, sz=0.38)
# Hem inferior del hoodie — ligero flare
cyl('hoodie_hem', 'hoodie', (0, 0, 1.22), r=0.68, h=0.18, segs=16)

# ── BRAZOS ────────────────────────────────────────────────────────────────────
# Cilindro con eje Z = vertical en Blender → correcto para brazos colgantes
cyl('armL', 'hoodie', (-1.02, 0, 1.95), r=0.20, h=1.15, segs=12)
cyl('armR', 'hoodie', ( 1.02, 0, 1.95), r=0.20, h=1.15, segs=12)

# ── MANOS ─────────────────────────────────────────────────────────────────────
sph('handL', 'skin', (-1.02, 0, 1.30), sx=0.24, sy=0.22, sz=0.22)
sph('handR', 'skin', ( 1.02, 0, 1.30), sx=0.24, sy=0.22, sz=0.22)

# ── PIERNAS ───────────────────────────────────────────────────────────────────
cyl('legL', 'pants', (-0.34, 0, 0.56), r=0.26, h=1.10, segs=12)
cyl('legR', 'pants', ( 0.34, 0, 0.56), r=0.26, h=1.10, segs=12)

# ── PIES ──────────────────────────────────────────────────────────────────────
cube('footL', 'shoes', (-0.34, 0.12, 0.10), sx=0.28, sy=0.46, sz=0.13)
cube('footR', 'shoes', ( 0.34, 0.12, 0.10), sx=0.28, sy=0.46, sz=0.13)

# ── Exportar GLB ──────────────────────────────────────────────────────────────
output = "C:/Users/denny/OneDrive/Escritorio/Proyectos Dennys/dennysh-portfolio/public/models/avatar.glb"
os.makedirs(os.path.dirname(output), exist_ok=True)

bpy.ops.object.select_all(action='SELECT')
bpy.ops.export_scene.gltf(
    filepath=output,
    export_format='GLB',
    export_apply=True,
    use_selection=False,
    export_materials='EXPORT',
    export_yup=True,
)

size_kb = os.path.getsize(output) / 1024
print(f"SUCCESS: {output}")
print(f"Size: {size_kb:.1f} KB")
```

Usar la herramienta Write para crear `src/components/Avatar3D.jsx` con este contenido:

```jsx
import { useRef, useEffect, Suspense } from 'react'
import { Canvas, useFrame } from '@react-three/fiber'
import { useGLTF } from '@react-three/drei'
import * as THREE from 'three'

// ── Toon gradient (4 tonos) ───────────────────────────────────────────────────
const toonGradient = (() => {
  const data = new Uint8Array([90, 148, 200, 238])
  const tex = new THREE.DataTexture(data, 4, 1, THREE.RedFormat)
  tex.magFilter = THREE.NearestFilter
  tex.needsUpdate = true
  return tex
})()

// ── Plataforma glow (geometría inline — separada del GLB) ─────────────────────
function Platform() {
  return (
    <group>
      {/* Disco principal */}
      <mesh>
        <cylinderGeometry args={[1.2, 1.2, 0.04, 48]} />
        <meshBasicMaterial color="#3B82F6" transparent opacity={0.85} />
      </mesh>
      {/* Ring exterior con glow difuso */}
      <mesh position={[0, -0.01, 0]}>
        <cylinderGeometry args={[1.65, 1.65, 0.02, 48]} />
        <meshBasicMaterial color="#3B82F6" transparent opacity={0.22} />
      </mesh>
    </group>
  )
}

// ── Modelo del avatar (cargado desde GLB) ─────────────────────────────────────
function AvatarModel({ mouseRef }) {
  const { scene } = useGLTF('/models/avatar.glb')
  const rootRef = useRef()
  const headGroupRef = useRef(null)
  const curRot = useRef({ x: 0, y: 0 })

  // Reemplazar materiales GLB (PBR) con MeshToonMaterial
  useEffect(() => {
    scene.traverse((child) => {
      if (child.isMesh && child.material) {
        const color = child.material.color?.clone() ?? new THREE.Color(0xffffff)
        child.material = new THREE.MeshToonMaterial({
          color,
          gradientMap: toonGradient,
        })
      }
    })

    // Buscar el empty 'head_group' para head tracking
    const hg = scene.getObjectByName('head_group')
    if (hg) headGroupRef.current = hg
  }, [scene])

  useFrame(({ clock }) => {
    const t = clock.getElapsedTime()

    // Float idle
    if (rootRef.current) {
      rootRef.current.position.y = Math.sin(t * 0.85) * 0.08
    }

    // Head tracking — muy sutil, con límites y suavizado lento
    if (headGroupRef.current && mouseRef.current) {
      const mx = mouseRef.current.x  // -1 a 1
      const my = mouseRef.current.y  // -1 a 1

      // Límites: ±15° en Y (yaw), ±8° en X (pitch)
      const targetY = Math.max(-0.26, Math.min(0.26,  mx * 0.18))
      const targetX = Math.max(-0.14, Math.min(0.14, -my * 0.10))

      // Lerp muy lento → movimiento orgánico
      curRot.current.x += (targetX - curRot.current.x) * 0.05
      curRot.current.y += (targetY - curRot.current.y) * 0.05

      headGroupRef.current.rotation.x = curRot.current.x
      headGroupRef.current.rotation.y = curRot.current.y
    }
  })

  // rotation Y 180°: Blender +Y → glTF -Z → necesita girar para mirar a cámara
  // position.y ajusta la altura del avatar en el canvas
  return <primitive ref={rootRef} object={scene} position={[0, -1.55, 0]} rotation={[0, Math.PI, 0]} />
}

// ── Fallback mientras carga el GLB ────────────────────────────────────────────
function AvatarFallback() {
  return (
    <mesh>
      <sphereGeometry args={[0.6, 8, 6]} />
      <meshBasicMaterial color="#1C1C26" transparent opacity={0.35} />
    </mesh>
  )
}

// ── Export principal ──────────────────────────────────────────────────────────
export default function Avatar3D() {
  const mouseRef = useRef({ x: 0, y: 0 })

  useEffect(() => {
    const onMove = (e) => {
      mouseRef.current = {
        x:  (e.clientX / window.innerWidth  - 0.5) * 2,
        y: -(e.clientY / window.innerHeight - 0.5) * 2,
      }
    }
    window.addEventListener('mousemove', onMove)
    return () => window.removeEventListener('mousemove', onMove)
  }, [])

  return (
    <Canvas
      camera={{ position: [0, 0.4, 5.5], fov: 45 }}
      dpr={[1, 1.5]}
      gl={{ antialias: true, alpha: true }}
      style={{ background: 'transparent' }}
    >
      <ambientLight intensity={0.7} />
      <directionalLight position={[2, 4, 5]}  intensity={1.4} color="#fff6e8" />
      <directionalLight position={[-2, 1, 2]} intensity={0.4} color="#7c3aed" />

      {/* Plataforma posicionada en la base del avatar */}
      <group position={[0, -1.8, 0]}>
        <Platform />
      </group>

      <Suspense fallback={<AvatarFallback />}>
        <AvatarModel mouseRef={mouseRef} />
      </Suspense>
    </Canvas>
  )
}

useGLTF.preload('/models/avatar.glb')
```
